    //Hidden:Lines
    private string canonicalURL { get; set; }

    protected override void OnInitialized()
    {
        canonicalURL = NavigationManager.Uri.Split("?")[0];
    }
    //End:Hidden
    /// <summary>
    /// Create and download the simple PDF document
    ///<summary>
    protected async void GeneratePdf(MouseEventArgs args)
    {	   
        XFAFormCreationService service = new XFAFormCreationService();
        MemoryStream documentStream = service.CreatePdfDocument();
        await JS.SaveAs("XFAFormCreation.pdf", documentStream.ToArray());
    }
}
