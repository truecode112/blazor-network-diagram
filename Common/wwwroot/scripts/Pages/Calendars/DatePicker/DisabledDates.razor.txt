<SampleDescription>
    <p>In the following example, all the weekends (<b>Saturday</b> and <b>Sunday</b>) of a month are <b>disabled</b>, and these dates are <b>restricted</b> to set or select in the DatePicker.</p>
</SampleDescription>
<ActionDescription>
    <p>Disabled Dates example demonstrates how to disable specific dates in the DatePicker by using <a href='https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Calendars.DatePickerEvents-1.html#Syncfusion_Blazor_Calendars_DatePickerEvents_1_OnRenderDayCell' target='_blank'>OnRenderDayCell</a> event. This event gets triggered on each day cell element creation, that allows you to customize, or disable specific dates in the DatePicker. Here the weekend dates are disabled by using <b>OnRenderDayCell</b>.</p>
</ActionDescription>

        <SfDatePicker TValue="DateTime?" CssClass="@CssClass" @ref="@DatePickerObj" Placeholder="Choose a Date" ShowClearButton="true">
            <DatePickerEvents TValue="DateTime?"  OnRenderDayCell="@DisableDate"></DatePickerEvents>

@code {
    private SfDatePicker<DateTime?> DatePickerObj { get; set; }
    public string CssClass { get; set; } = "e-popup-expand";
    //Hidden:Lines
    private string canonicalURL { get; set; }
    protected override void OnInitialized()
    {
        canonicalURL = NavigationManager.Uri.Split("?")[0];
    }
    //End:Hidden
    private void DisableDate(RenderDayCellEventArgs args)
    {
        var view = DatePickerObj.CurrentView();
        if (view == "Month" && ((int)args.Date.DayOfWeek == 0 || (int)args.Date.DayOfWeek == 6))
        {
            args.IsDisabled = true;
        }
    }
}
